absval <- function(x) {
    if (x>=0){ # return x if positive or zero
        y=x
    } else { # if negative, return the positive version
        y=-x
    }
}

# check that the function works

print(absval(0))
print(absval(-3))

# plot the function for some points
x = seq(-5,5,0.5)

    y[n] = absval(x[n])
}

print(y)

plot(x,y)
